    function toggleMenu() {
        const menu = document.getElementById('menu');

        const toggleButton = document.getElementById('toggleButton');
        menu.classList.toggle('open');
        toggleButton.classList.toggle('open');

            menu.style.left = '0';
        } else {
            menu.style.left = '-220px';
        }
